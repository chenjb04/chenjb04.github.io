# 定义一个函数来计算从1加到n的总和  
def sum_to_n(n):  
    total = 0  
    for i in range(1, n + 1):  
        total += i  
    return total  
  
# 从用户那里获取输入  
n = int(input("请输入一个数字: "))  
  
# 调用函数并打印结果  
print(f"从1加到{n}的总和是: {sum_to_n(n)}")
